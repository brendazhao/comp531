const validateAccountName=(element)=>{
    var reg=/^([a-zA-Z]{1})([a-zA-Z0-9]{1})+$/;
    if(element.value.match(reg)||!(validate(element))){  
            return true; 
        }
        return false;     
}

const validateEmail=(element)=>{

            return true; 
        }
        return false;     
}
